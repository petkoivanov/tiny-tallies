const SUBSCRIPTS: Record<number, string> = {
  0: '\u2080', 1: '\u2081', 2: '\u2082', 3: '\u2083', 4: '\u2084',

`generateLog10Eval(template, rng)`:
- answer = rng.intRange(1, 6). argument = Math.pow(10, answer).
- wrongArgument = argument (student gave argument instead of exponent).
- wrongOffByOne = answer + 1 (miscounted zeros).
- formattedArg = argument.toLocaleString() (adds commas for 1,000+).
- questionText: `Evaluate log${SUBSCRIPTS[1]}${SUBSCRIPTS[0]}(${formattedArg}).`
- operands: [wrongArgument, wrongOffByOne, answer].
- return numericAnswer(answer).

`generateLog2Eval(template, rng)`:
- answer = rng.intRange(1, 10). argument = Math.pow(2, answer).
- wrongArgument = argument.
- wrongOffByOne = answer - 1 (off-by-one).
- questionText: `Evaluate log${SUBSCRIPTS[2]}(${argument}).`
- operands: [wrongArgument, wrongOffByOne, answer].
- return numericAnswer(answer).

`generateLnEval(template, rng)`:
- answer = rng.intRange(1, 5).
- sup = SUPERSCRIPTS[answer] ?? `^${answer}` (handles answer=1 which has no superscript entry: use `^1`; but better to handle 1 as special case: if answer === 1, questionText is "Evaluate ln(e)." with no superscript).
- wrongSquared = answer * answer (confused ln(e^n) with n^2).
- wrongOffByOne = answer + 1.
- questionText: answer === 1 ? `Evaluate ln(e).` : `Evaluate ln(e${sup}).`
- CRITICAL: Never display the numeric value of e^n. Always display as "ln(e)" or "ln(e^n)".
- operands: [wrongSquared, wrongOffByOne, answer].
- return numericAnswer(answer).

`generateLogWordProblem(template, rng)`:
- Reuse generateLog10Eval logic -- the word problem system's prefix mode will prepend context.
- Same implementation as generateLog10Eval.
