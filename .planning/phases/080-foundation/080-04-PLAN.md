    ```typescript
    import React, { useState } from 'react';
    import { View, Text, Pressable, StyleSheet } from 'react-native';
    import { setsEqual } from '@/services/mathEngine/types';
    import type { ChoiceOption } from '@/services/mathEngine/answerFormats/types';
    import { useAppStore } from '@/store/appStore';

    interface MultiSelectMCProps {
      options: readonly ChoiceOption[];
      correctIndices: readonly number[];
      onAnswer: (correct: boolean) => void;
    }

    export default function MultiSelectMC({ options, correctIndices, onAnswer }: MultiSelectMCProps) {
      const [selectedIndices, setSelectedIndices] = useState<Set<number>>(new Set());
      const [submitted, setSubmitted] = useState(false);
      const theme = useAppStore((s) => s.theme); // for colors

      function toggleOption(index: number) {
        if (submitted) return;
        setSelectedIndices((prev) => {
          const next = new Set(prev);
          if (next.has(index)) { next.delete(index); } else { next.add(index); }
          return next;
        });
      }

      function handleCheck() {
        if (selectedIndices.size === 0 || submitted) return;
        const selected = Array.from(selectedIndices);
        const correct = setsEqual(selected, [...correctIndices]);
        setSubmitted(true);
        onAnswer(correct);
      }

      // ... render options as Pressables with checkboxes
      // ... render Check button (disabled when selectedIndices.size === 0 || submitted)
    }
    ```
