    ## Problem Selector (problemSelector.ts)
    - TARGET_SUCCESS = 0.85, SIGMA = 0.10 (exported constants)
    - weightBySuccessProbability(studentElo, templates, targetSuccess?): returns WeightedTemplate[]
      - For each template: compute expected success via expectedScore(studentElo, template.baseElo)
      - Weight = exp(-((p - target)^2) / (2 * sigma^2)) -- gaussian weighting on probability deviation
      - Templates near 85% success get weight near 1.0
      - Templates far from 85% get weight near 0.0
    - weightedRandomSelect(weighted, rng): returns ProblemTemplate
      - Cumulative distribution sampling: sum weights, roll rng.next() * totalWeight, pick first item where cumulative exceeds roll
      - Fallback to last item if floating point edge case
    - selectTemplateForSkill(skillId, studentElo, rng, frustrationActive?): returns ProblemTemplate
      - Gets templates via getTemplatesBySkill(skillId)
      - If frustrationActive: filter to templates with baseElo < studentElo; if all templates have baseElo >= studentElo, pick the one with lowest baseElo
      - Compute weights via weightBySuccessProbability
      - Select via weightedRandomSelect
      - Throw if no templates found for skill

    ## Skill Selector (skillSelector.ts)
    - WEAKNESS_BASELINE = 50 (exported constant)
    - weightSkillsByWeakness(unlockedSkillIds, skillStates, defaultElo?): returns SkillWeight[]
      - For each skill: get Elo from skillStates (or defaultElo if absent)
      - Find maxElo across all unlocked skills
      - Weight = (maxElo - skillElo) + WEAKNESS_BASELINE
      - All skills get non-zero weight (baseline ensures it)
      - Weakest skill gets highest weight
    - selectSkill(unlockedSkillIds, skillStates, rng, defaultElo?): returns string
      - Get weights via weightSkillsByWeakness
      - Weighted random select (same cumulative distribution approach)
      - Returns selected skillId

         ```typescript
         export function weightBySuccessProbability(
           studentElo: number,
           templates: readonly ProblemTemplate[],
           targetSuccess: number = TARGET_SUCCESS,
         ): WeightedTemplate[] {
           return templates.map(template => {
             const p = expectedScore(studentElo, template.baseElo);
             const deviation = p - targetSuccess;
             const weight = Math.exp(-(deviation * deviation) / (2 * SIGMA * SIGMA));
             return { template, weight, expectedSuccess: p };
           });
         }
         ```
       - Implement `weightedRandomSelect`:
         ```typescript
         export function weightedRandomSelect(
           weighted: readonly WeightedTemplate[],
           rng: SeededRng,
         ): ProblemTemplate {
           const totalWeight = weighted.reduce((sum, w) => sum + w.weight, 0);
           let roll = rng.next() * totalWeight;
           for (const w of weighted) {
             roll -= w.weight;
             if (roll <= 0) return w.template;
           }
           return weighted[weighted.length - 1].template;
         }
         ```
       - Implement `selectTemplateForSkill`:
         ```typescript
         export function selectTemplateForSkill(
           skillId: string,
           studentElo: number,
           rng: SeededRng,
           frustrationActive: boolean = false,
         ): ProblemTemplate {
           let templates = getTemplatesBySkill(skillId);
           if (templates.length === 0) {
             throw new Error(`No templates found for skill: ${skillId}`);
           }

           if (frustrationActive) {
             const easier = templates.filter(t => t.baseElo < studentElo);
             if (easier.length > 0) {
               templates = easier;
             } else {
               // All templates are at or above student Elo; pick the easiest one
               templates = [templates.reduce((a, b) => a.baseElo <= b.baseElo ? a : b)];
             }
           }

           const weighted = weightBySuccessProbability(studentElo, templates);
           return weightedRandomSelect(weighted, rng);
         }
         ```

         ```typescript
         export function weightSkillsByWeakness(
           unlockedSkillIds: readonly string[],
           skillStates: Record<string, SkillState>,
           defaultElo: number = 1000,
         ): SkillWeight[] {
           const elos = unlockedSkillIds.map(id => ({
             skillId: id,
             elo: skillStates[id]?.eloRating ?? defaultElo,
           }));
           const maxElo = Math.max(...elos.map(e => e.elo));
           return elos.map(({ skillId, elo }) => ({
             skillId,
             elo,
             weight: maxElo - elo + WEAKNESS_BASELINE,
           }));
         }
         ```

         ```typescript
           skillStates: Record<string, SkillState>,

           defaultElo: number = 1000,
         ): string {
           const weights = weightSkillsByWeakness(unlockedSkillIds, skillStates, defaultElo);

         }
         ```
