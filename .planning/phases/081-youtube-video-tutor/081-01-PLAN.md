      it('initializes videoVotes as empty object', () => {

      it('setVideoVote overwrites a previous vote', () => {

        setVideoVote('fractions' as MathDomain, 'not_helpful');
        expect(useAppStore.getState().videoVotes['fractions']).toBe('not_helpful');
