    - "007" -> 7 (strips leading zeros)
    - "0" -> 0 (preserves zero itself)
    - "" -> null (empty string)
    - "abc" -> null (non-numeric)
    - "3.14" -> null (decimals rejected)
    - "-5" -> null (negative signs rejected)
    - "12 34" -> null (internal spaces rejected)
    - "99999" -> null (exceeds 9999 bound)
    - "9999" -> 9999 (at bound, accepted)

    ## validateFreeTextAnswer(input: string, correctAnswer: number) -> { correct: boolean; parsedValue: number | null }
    - "42" with correctAnswer 42 -> { correct: true, parsedValue: 42 }
    - "43" with correctAnswer 42 -> { correct: false, parsedValue: 43 }
    - "abc" with correctAnswer 42 -> { correct: false, parsedValue: null }
    - "  042  " with correctAnswer 42 -> { correct: true, parsedValue: 42 }

      3. Strip leading zeros (keep "0" itself): `const normalized = trimmed.replace(/^0+(?=\d)/, '')`

      6. Reject > 9999: `if (value > 9999) return null`
      7. Return value

      3. Return `{ correct: parsed === correctAnswer, parsedValue: parsed }`
