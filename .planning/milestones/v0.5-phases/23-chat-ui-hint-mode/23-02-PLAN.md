    ```typescript
    const handleResponse = (type: 'understand' | 'more' | 'confused' | 'retry') => {
      switch (type) {
        case 'understand': {
          addTutorMessage({
            id: `child-${Date.now()}`,
            role: 'child',
            text: "I understand!",
            timestamp: Date.now(),
          });
          // Show encouragement then auto-close
          setTimeout(() => {
            addTutorMessage({
              id: `tutor-encourage-${Date.now()}`,
              role: 'tutor',
              text: "Great! Give it a try!",
              timestamp: Date.now(),
            });
          }, 200);
          autoCloseTimerRef.current = setTimeout(() => {
            setChatOpen(false);
          }, 1500);
          break;
        }
        case 'more': {
          addTutorMessage({
            id: `child-${Date.now()}`,
            role: 'child',
            text: "Tell me more",
            timestamp: Date.now(),
          });
          tutor.requestHint(); // This increments hint level via useTutor
          break;
        }
        case 'confused': {
          addTutorMessage({
            id: `child-${Date.now()}`,
            role: 'child',
            text: "I'm confused",
            timestamp: Date.now(),
          });
          // Request hint at same level (useTutor will generate a rephrased response)
          // Do NOT increment hint level -- confused signals struggle, not progression
          tutor.requestHint();
          break;
        }
        case 'retry': {
          tutor.requestHint();
          break;
        }
      }
    };
    ```

    IMPORTANT for "I'm confused": The context says "for now triggers a rephrased hint." Since `requestHint()` calls `incrementHintLevel()` internally after a successful response, "I'm confused" will still get a new hint at a slightly higher level. This is acceptable -- the LLM non-determinism provides different phrasing. Phase 24 will add proper escalation logic for "I'm confused."
